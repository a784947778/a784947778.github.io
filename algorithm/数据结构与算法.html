/**
* 从第一位开始循环，与当前索引之后的比较，得到最小的值与当前位置交换，可能会存在相同值位置交换问题，不稳
*/
static void selectionSort(int[] arr) {
    for(int i=0; i<arr.length-1; i++){
        int minPos = i;
        for(int j=i+1; i<arr.length; j++){
            minPos = arr[j] < arr[minPos] ? j : minPos;
        }
        swap(arr,i,minPos);
    }
}

        for(int j = i;j > 0 && arr[j] < arr[j-1];j--) {

# 堆排序：

/**
* 先把数组按照大根堆或者小根堆的信息排序,然后逐个从树头(或者数组的头部)弹出最大值，每弹出一个heapSize减一，然后把内部结构再次以大根堆或者小根堆的形
* 式排序
*/
static void heapSort(int[] arr){
    if (arr == null || arr.length < 2) {return;}
    for (int i = 0; i < arr.length; i++) { // O(N)
        heapInsert(arr,i); // O(logN)
    }
    int heapSize = arr.length;
    swap(arr,0,--heapSize);
    while (heapSize > 0) {
        heapify(arr,0,heapSize); // O(logN)
        swap(arr,0,--heapSize); // O
    }
}
/**
* (index - 1) / 2子节点计算父节点索引
*/
static void heapInsert(int[] arr,int index){
    int f;
    while (arr[index] > arr[f = (index - 1) / 2]) {
       	// 父节点和子节点换位置
        swap(arr,index,f);
        index = f;
    }
}
static void heapify(int[] arr,int index,int heapSize){
    int left = index * 2 + 1;
    while (left < heapSize) {
        // 判断左右节点的大小，获取值大的节点
        int largest = left + 1 < heapSize && arr[left + 1] > arr[left] ? left + 1 : left;
        // 判断父节点和子节点的大小
        largest = arr[largest] > arr[index] ? largest : index;
        if (largest == index) {break;}
        swap(arr,largest,index);
        index = largest;
        left = index * 2 + 1;
    }
}

static void shellSort(int[] arr){
    int h = 1;
    while(h <= arr.length/3){h = h*3 + 1;}
    for(int gap = h; gap > 0; gap = (gap-1)/3){
        for(int i = gap; i < arr.length; i++){
            for(int j = 1; j>gap - 1;j -= gap){
                swap(arr,j,j-gap);
            }
        }
    }
}

# 归并排序：

/**
 * 数组递归二分，然后归并排序，依次返回
 */
static void sort(int[] arr,int left,int right){
    // 如果数组为null不需要排序， 如果left和right相同不能再拆分了
    if (arr == null || left == right) {
        return;
    }
    int mid = left + (right-left)/2;
    // 二分以后的数组继续拆分
    sort(arr,left,mid);
    sort(arr,mid+1,right);
    // 对拆分的左右不分进行归并排序
    mergeSort(arr,left,mid+1,right);
}
/**
 * 对指定部分归并排序
 */
static void mergeSort(int[] arr,int left,int right,int rightBound){
    int mid = right - 1;
    // 临时记录排好序的数据
    int[] temp = new int[rightBound - left + 1];
    int i = left; int j = right; int k = 0;
    // 比较排序
    while(i <= mid && j <= rightBound) 
		temp[k++] = arr[i] <= arr[j] ? arr[i++] : arr[j++];
    // 如果比较结束，左面部分或者右面部分还有数据，直接放入到临时数组中
    while(i <= mid) temp[k++] = arr[i++];
    while(j <= rightBound) temp[k++] = arr[j++];
    // 把排好序的数组放入到原数组中
    for(int m = 0;m < temp.length; m++) arr[left+m] = temp[m];
}

static void sort(int[] arr,int leftBound,int rightBound){
    if(leftBound >= rightBound) return;
    int mid = quickSort(arr,leftBound,rightBound);
    sort(arr,leftBound,mid-1);
    sort(arr,mid+1,rightBound);
}
static int quickSort(int[] arr,int leftBound,int rightBound){
    int pivot = arr[rightBound];
    int left = leftBound;
    int right = rightBound - 1;
    while(left <= right){
        while(left <= right && arr[left] <= pivot) left++;
        while(right >= left && arr[right] > pivot) right--;
        if(left < right) swap(arr,left,right);// TODO 判断是否多余
    }
    swap(arr,left,rightBound);
    return left;
}

# 计数排序：

/**
* 非比较排序，适合量大但是范围小的排序，创建一个count数组，长度为范围的最大值,遍历目标数组，在count对应的索引位置++，统计思想,当前实现方法对于引用
* 类型排序
*/
static int[] countingSort(int[] arr){
    int[] result = new int[arr.length];
    int[] count = new int[10];
    for(int i = 0;i < arr.length;i++) {count[arr[i]]++;}
    //这样实现不稳定
    //for(int i = 0,j = 0;i < count.length;i++) {
    //  while(count[i]-- > 0)result[j++] = i;
    //}
    //实现方法的稳定
    for(int i = 1;i < count.length;i++) {count[i] += count[i - 1];}
    //倒序目标数组，查询count数组对应的应该存放的位置，对应位置-1，可以实现排序的稳定性
    for(int i = arr.length-1;i > 0;i--) {
        result[--count[arr[i]]] = arr[i];
    }
    return result;
}

static void radixSort(){
    if (arr == null || arr.length < 2) {return;}

}
static int maxbits(int[] arr) {
    int max = Integer.MIN_VALUE;
    for (int i = 0;i < arr,length; i++) {max = Math.max(max,arr[i])}
    int res = 0;
    while (max != 0) {res++; max /= 10;}
    return res;
}
static void radixSort(int[] arr,int L,int R,int digit) {
    final int radix = 10;
    int i = 0; j = 0;
    // 有多少数准备多少辅助空间
    int[] bucket = new int[R - L + 1];
    for (int d = 1; d <= digit; d++) { // 有多少位就进出几次
        int[] count = new int[radix];
        for (i = L; i <= R; i++) {
            j = getDigit(arr[i],d); count[j]++;
        }
        for (i = 1; i < radix; i++) {
            count[i] = count[i] + count[i - 1]；
        }
        for (i = R; i >= L; i--) {
            j = getDigit(arr[i],d);
            bucket[count[j] - 1] = arr[i];
            count[j]--;
        }
        for (i = L,j = 0; i <= R; i++, j++) {
            arr[i] = bucket[j];
        }
    }        
}
static int getDigit(int x, int d) {
    return ((x / ((int) Math.pow(10, d - 1))) % 10);
}

/**
 * 公共前后缀实现加速
 */
public static int kmp(String s1, String s2) {
    // 长度不够肯定不包含
    if (s1 == null || s2 == null || s2.length() < 1 || s1.length() < s2.length()) {
        return -1;
    }
    int[] nextArr = nextArr(s2);
    int i = 0;
    int j = 0;
    while (i < s1.length() && j < s2.length()) {
        if (s1.charAt(i) == s2.charAt(j)) {
            i++;
            j++;
        } else if (j == 0) {
            i++;
        } else {
            j = nextArr[j];
        }
    }
    return j == s2.length() ? i - j : -1;
}

/**
 * 计算每一位字符之前的字符串的公共前后缀
,*/
public static int[] nextArr(String s2) {
        int[] next = new int[s2.length()];
        char[] chars = s2.toCharArray();
        next[0] = -1;
        next[1] = 0;
        // 从第二个开始计算
        int i = 2;
        /**
         * 当前cn有两个意思
         * 1、i上一个字符之前的最长公共前缀
         * 2、公共前缀+1的索引
         */
        int cn = 0;
        while (i < s2.length()) {
            if (chars[i - 1] == chars[cn]) {
                // 如果相同，在上一个字符之前字符串的公共前缀长度的基础上加1
                // 当前的为最新字符的公共前缀长度
                next[i++] = ++cn;
            } else if (cn > 0) {
                // 如果不相同，且上一个字符之前字符串的公共前缀长度大于0
                cn = next[cn];
            } else {
                // 所用当前字符之前的字符串没有公共前后缀，计算下一个字符的公共前后缀
                next[i++] = 0;
            }
        }
        return next;
    }

/**
 * 深度优先搜索（DFS）
 * 注意：不能用队列实现容易有漏掉的节点
 */
public void dfs(Node node) {
    Deque<Node> deque = new LinkedList<>();
    Set<Node> set = new HashSet<>();

    deque.add(node);
    set.add(node);
    // to do something
    System.out.println(node.value);
    while (!deque.isEmpty()) {
        Node poll = deque.pop();
        for (Node next : poll.nexts) {
            if (!set.contains(next)) {
                deque.add(poll);
                deque.add(next);
                set.add(next);
                // to do something
                System.out.println(next.value);
                break;
            }
        }
    }
}

# 广度优先遍历（bfs）：

 /**
  * 广度优先搜索（BFS）
  */
public void bfs(Node node) {
    Queue<Node> queue = new LinkedList<>();
    Set<Node> set = new HashSet<>();

    queue.add(node);
    set.add(node);
    while (!queue.isEmpty()) {
        Node poll = queue.poll();
        // to do something
        System.out.println(node.value);
        List<Node> nexts = poll.nexts;
        for (Node next : nexts) {
            if (!set.contains(node)) {
                queue.add(next);
                set.add(next);
            }
        }
    }
}

# 拓扑排序（有向图）：已入度为0的顶点开始，依次删除，不能有环，并删除关联顶点的入度，按照依赖的关系，依次排序

public List<Node> topologySort(Graph graph) {
    Map<Integer, Node> nodes = graph.nodes;
    // 不在原有数据修改，需要空间暂存变动的数据
    Map<Node, Integer> inMap = new HashMap<>();
    Queue<Node> zeroInQueue = new LinkedList<>();
    // 入读为0的点放入到队列中
    for (Map.Entry<Integer, Node> entry : nodes.entrySet()) {
        if (entry.getValue().in == 0) {
            zeroInQueue.add(entry.getValue());
        }
    }

    // 拓扑排序的最终结果
    List<Node> result = new ArrayList<>();
    while(!zeroInQueue.isEmpty()) {
        Node cur = zeroInQueue.poll();
        result.add(cur);
        // 所有next节点的入度减1
        for (Node next : cur.nexts) {
            inMap.put(next, inMap.getOrDefault(next, next.in) - 1);
            if (next.in == 0) {
                // 如果入度为0，加入队列
                zeroInQueue.add(next);
            }
        }
    }
    return result;
}

    2. 快指针走1步，慢指针走1步，当再次相遇的时候，一定是在相交的点。

public static int binarySearch(Integer[] srcArray, int des) {
    //定义初始最小、最大索引
    int start = 0;
    int end = srcArray.length - 1;
    //确保不会出现重复查找，越界
    while (start <= end) {
        //计算出中间索引值
        int middle = (end + start)>>>1 ;//防止溢出
        if (des == srcArray[middle]) {
            return middle;
        //判断下限
        } else if (des < srcArray[middle]) {
            end = middle - 1;
        //判断上限
        } else {
            start = middle + 1;
        }
    }
    //若没有，则返回-1
    return -1;
}
