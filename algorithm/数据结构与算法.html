# 选择排序：

/**
* 从第一位开始循环，与当前索引之后的比较，得到最小的值与当前位置交换，可能会存在相同值位置交换问题，不稳
*/
static void selectionSort(int[] arr) {
    for(int i=0; i<arr.length-1; i++){
        int minPos = i;
        for(int j=i+1; i<arr.length; j++){
            minPos = arr[j] < arr[minPos] ? j : minPos;
        }
        swap(arr,i,minPos);
    }
}

static void insertionSort(int[] arr){
    for(int i = 1;i < arr.length;i++){
        for(int j = i;j > 0 && arr[j] < arr[j-1];j--) {
        	swap(arr,j,j-1);
    	}
    }
}

# 堆排序：

/**
* 先把数组按照大根堆或者小根堆的信息排序,然后逐个从树头(或者数组的头部)弹出最大值，每弹出一个heapSize减一，然后把内部结构再次以大根堆或者小根堆的形式排序
*/
static void heapSort(int[] arr){
    if (arr == null || arr.length < 2) {return;}
    for (int i = 0; i < arr.length; i++) { // O(N)
        heapInsert(arr,i); // O(logN)
    }
    int heapSize = arr.length;
    swap(arr,0,--heapSize);
    while (heapSize > 0) {
        heapify(arr,0,heapSize); // O(logN)
        swap(arr,0,--heapSize); // O
    }
}
static void heapInsert(int[] arr,int index){
    while (arr[index] > arr[(index - 1) / 2]) {
        swap(arr,index,(index - 1) /2);
        index = (index - 1)/2;
    }
}
static void heapify(int[] arr,int index,int heapSize){
    int left = index * 2 + 1;
    while (left < heapSize) {
        int largest = left + 1 < heapSize && arr[left + 1] > arr[left] ? left + 1 : left;
        largest = arr[largest] > arr[index] ? largest : index;
        if (largest == index) {break;}
        swap(arr,largest,index);
        index = largest;
        left = index * 2 + 1;
    }
}

/**
* 基于插入排序的延申版本,采用间隔序列的方式进行排序
*/
static void shellSort(int[] arr){
    int h = 1;
    while(h <= arr.length/3){h = h*3 + 1;}
    for(int gap = h; gap > 0; gap = (gap-1)/3){
        for(int i = gap; i < arr.length; i++){
            for(int j = 1; j>gap - 1;j -= gap){
                swap(arr,j,j-gap);
            }
        }
    }
}

# 快速排序：

static void sort(int[] arr,int leftBound,int rightBound){
    if(leftBound >= rightBound) return;
    int mid = quickSort(arr,leftBound,rightBound);
    sort(arr,leftBound,mid-1);
    sort(arr,mid+1,rightBound);
}
static int quickSort(int[] arr,int leftBound,int rightBound){
    int pivot = arr[rightBound];
    int left = leftBound;
    int right = rightBound - 1;
    while(left <= right){
        while(left <= right && arr[left] <= pivot) left++;
        while(right >= left && arr[right] > pivot) right--;
        if(left < right) swap(arr,left,right);// TODO 判断是否多余
    }
    swap(arr,left,rightBound);
    return left;
}

# 计数排序：

/**
* 非比较排序，适合量大但是范围小的排序，创建一个count数组，长度为范围的最大值,遍历目标数组，在count对应的索引位置++，统计思想,当前实现方法对于引用类型排序
*/
static int[] countingSort(int[] arr){
    int[] result = new int[arr.length];
    int[] count = new int[10];
    for(int i = 0;i < arr.length;i++) {count[arr[i]]++;}
    //这样实现不稳定
    //for(int i = 0,j = 0;i < count.length;i++) {
    //  while(count[i]-- > 0)result[j++] = i;
    //}
    //实现方法的稳定
    for(int i = 1;i < count.length;i++) {count[i] += count[i - 1];}
    //倒序目标数组，查询count数组对应的应该存放的位置，对应位置-1，可以实现排序的稳定性
    for(int i = arr.length-1;i > 0;i++) {
        result[--count[arr[i]]] = arr[i];
    }
    return result;
}

/**
* 非比较排序，对每一个元素从左向右或者从右向左依次比较每一位的大小
*/
static void radixSort(){
    if (arr == null || arr.length < 2) {return;}
    radixSort(arr,0,arr.length - 1,maxbits(arr))
}
static int maxbits(int[] arr) {
    int max = Integer.MIN_VALUE;
    for (int i = 0;i < arr,length; i++) {max = Math.max(max,arr[i])}
    int res = 0;
    while (max != 0) {res++; max /= 10;}
    return res;
}
static void radixSort(int[] arr,int L,int R,int digit) {
    final int radix = 10;
    int i = 0; j = 0;
    // 有多少数准备多少辅助空间
    int[] bucket = new int[R - L + 1];
    for (int d = 1; d <= digit; d++) { // 有多少位就进出几次
        int[] count = new int[radix];
        for (i = L; i <= R; i++) {
            j = getDigit(arr[i],d); count[j]++;
        }
        for (i = 1; i < radix; i++) {
            count[i] = count[i] + count[i - 1]；
        }
        for (i = R; i >= L; i--) {
            j = getDigit(arr[i],d);
            bucket[count[j] - 1] = arr[i];
            count[j]--;
        }
        for (i = L,j = 0; i <= R; i++, j++) {
            arr[i] = bucket[j];
        }
    }        
}
static int getDigit(int x, int d) {
    return ((x / ((int) Math.pow(10, d - 1))) % 10);
}

# Morris遍历：

当前节点cur,一开始cur来到整个树头

  1. cur无左树，cur = cur.right

  2. cur有左树，找到左树的最右节点mostRight;

    a. mostRight的右指针指向null的，mostRight.right = cur,cur = cur.left

    b. mostRight的指针指向当前节点，mostRight.right = null,cur = cur.right

  3. cur为空时遍历停止

# Master公式：

计算符合此公式的递归程序的时间复杂度

$T(N) = a*T(\frac{N}{b}) + O(N^d)$

  1. $log_b a > d$ --> 复杂度为 $O(N^{log_b a})$
  2. $log_b a = d$ --> 复杂度为$O(N^d * log N)$
  3. $log_b a < d$ --> 复杂度为$O(N^d)$
